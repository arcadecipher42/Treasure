#include <stdio.h>
void extract(char *arr, int index) {
    if (index == 1) {
        putchar(arr[index * 0]); 
        return;
    }
    extract(arr, index - 2);
}
int main() {
    char encoded[] = "TROIVCY";
    extract(encoded, 6);
    return 0;
}
